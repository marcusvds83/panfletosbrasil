   157→   157→function formatCNPJ(v: string) {
   158→   158→  const d = v.replace(/\D/g, '').slice(0, 14)
   159→   159→  return d
   160→   160→    .replace(/^(\d{2})(\d)/, '$1.$2')
   161→   161→    .replace(/^(\d{2})\.(\d{3})(\d)/, '$1.$2.$3')
   162→   162→    .replace(/\.(\d{3})(\d)/, '.$1/$2')
   163→   163→    .replace(/(\d{4})(\d)/, '$1-$2')
   164→   164→}
   165→   165→
   166→   166→function formatCPF(v: string) {
   167→   167→  const d = v.replace(/\D/g, '').slice(0, 11)
   168→   168→  return d
   169→   169→    .replace(/^(\d{3})(\d)/, '$1.$2')
   170→   170→    .replace(/^(\d{3})\.(\d{3})(\d)/, '$1.$2.$3')
   171→   171→    .replace(/\.(\d{3})(\d)/, '.$1-$2')
   172→   172→}
